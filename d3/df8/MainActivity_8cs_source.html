299  private string GetPathToImage(Android.Net.Uri uri)
300  {
301  string doc_id = "";
302  using (var c1 = ContentResolver.Query(uri, null, null, null, null))
303  {
304  c1.MoveToFirst();
305  String document_id = c1.GetString(0);
306  doc_id = document_id.Substring(document_id.LastIndexOf(":") + 1);
307  }
308 
309  string path = null;
310 
311  // The projection contains the columns we want to return in our query.
312  string selection = Android.Provider.MediaStore.Images.Media.InterfaceConsts.Id + " =? ";
313  using (var cursor = ManagedQuery(Android.Provider.MediaStore.Images.Media.ExternalContentUri, null, selection, new string[] { doc_id }, null))
314  {
315  if (cursor == null) return path;
316  var columnIndex = cursor.GetColumnIndexOrThrow(Android.Provider.MediaStore.Images.Media.InterfaceConsts.Data);
317  cursor.MoveToFirst();
318  path = cursor.GetString(columnIndex);
319  }
320  return path;
321  }
