267  private string GetPathToImage(Android.Net.Uri uri)
268  {
269  string doc_id = "";
270  using (var c1 = ContentResolver.Query(uri, null, null, null, null))
271  {
272  c1.MoveToFirst();
273  String document_id = c1.GetString(0);
274  doc_id = document_id.Substring(document_id.LastIndexOf(":") + 1);
275  }
276 
277  string path = null;
278 
279  // The projection contains the columns we want to return in our query.
280  string selection = Android.Provider.MediaStore.Images.Media.InterfaceConsts.Id + " =? ";
281  using (var cursor = ManagedQuery(Android.Provider.MediaStore.Images.Media.ExternalContentUri, null, selection, new string[] { doc_id }, null))
282  {
283  if (cursor == null) return path;
284  var columnIndex = cursor.GetColumnIndexOrThrow(Android.Provider.MediaStore.Images.Media.InterfaceConsts.Data);
